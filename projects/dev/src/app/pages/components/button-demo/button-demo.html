Usage Code:

<button
    zButton
    color="color()"
    size="{{ size() }}"
    style="{{ style() }}"
    {{ shape() ? 'shape="' + shape() + '"' : '' }}
    {{ width() ? 'width="' + width() + '"' : '' }}
    animation="{{ animation() }}"
    {{ loading() ? '[loading]="true"' : '' }}
    {{ disabled() ? '[disabled]="true"' : '' }}
    {{ glass() ? '[glass]="true"' : '' }}
    {{ noAnimation() ? '[noAnimation]="true"' : '' }}
    >
    {{ buttonText() }}
    </button>
